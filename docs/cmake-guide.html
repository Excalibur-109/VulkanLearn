CMake 代码编辑大全

以 现代 CMake（3.20+） 为基础，重点是写出 清晰、可移植、不坑人 的 CMakeLists.txt。
风格选择：以 Visual Studio Generator 为主示例，但所有思想跨 generator 都成立。

1. CMake 是什么

CMake 是「构建系统生成器」。它本身不编译，只是根据 CMakeLists.txt 生成底层构建系统的工程文件：

  • Visual Studio → .sln / .vcxproj
  • Ninja → build.ninja
  • Make → Makefile
  • Xcode → .xcodeproj

两阶段：

  1. Configure：解析 CMakeLists.txt，生成构建文件（输出 build/）
  2. Build：底层构建系统（MSBuild / Ninja 等）执行编译链接

2. CMakeLists.txt 骨架

# 1. 最低版本 + 项目声明
cmake_minimum_required(VERSION 3.20)
project(MyApp VERSION 0.1.0 LANGUAGES CXX)

# 2. 通用设置
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

if(MSVC)
    add_compile_options(/W4 /permissive- /Zc:__cplusplus)
    add_compile_definitions(NOMINMAX)
endif()

# 3. 找依赖
find_package(Threads REQUIRED)
find_package(fmt CONFIG REQUIRED)

# 4. 定义 target
add_executable(MyApp main.cpp)

# 5. 设置 target 的 include、compile、link
target_link_libraries(MyApp
    PRIVATE fmt::fmt Threads::Threads
)
现代 CMake 的核心思想：用 target_* 命令把属性绑定到具体的 target 上，而不是用全局 include_directories、add_definitions 等。

3. 变量与缓存

CMake 的变量都是字符串，没有真正的类型。

# 普通变量
set(MY_NAME "hello")
set(LIST a b c)            # a; b; c
set(LIST "a;b;c")          # 同样

# 引用
message(STATUS "name=${MY_NAME}")
message(STATUS "list=${LIST}")  # a;b;c

# 列表操作
list(LENGTH LIST n)
list(APPEND LIST d)
list(REMOVE_ITEM LIST a)
list(SORT LIST)

# 字符串操作
string(TOUPPER "abc" X)
string(REPLACE "a" "b" X "aXa")  # X=bXb
string(REGEX MATCH "[0-9]+" N "abc123")

# 环境变量
set(VULKAN_SDK_FROM_ENV "$ENV{VULKAN_SDK}")

缓存变量（写到 CMakeCache.txt，GUI 或 cmake-gui 看得到）：

# 缓存 + 类型限定
set(BUILD_TESTS ON CACHE BOOL "Build unit tests")

# 等价 option：option 实际就是 set(... BOOL CACHE ...) 加一层封装
option(ENABLE_ASAN "Enable AddressSanitizer" OFF)
缓存变量一旦在 GUI 中改过一次，后续每次 configure 都会保留这个值，即便源码里改成 OFF 也不变。要重置就删 CMakeCache.txt。

4. 流程控制

# if
if(MINGW)
    ...
elseif(MSVC)
    ...
else()
    ...
endif()

# 常用比较
# VAR                          - 真假判定
# STREQUAL "x"                 - 字符串相等
# VERSION_LESS 1.2 1.3         - 版本
# EQUAL 1 2 / LESS 1 2         - 数字
# DEFINED                      - 变量是否定义
# EXISTS path                  - 文件存在
# IS_DIRECTORY / IS_ABSOLUTE   - 路径类型
# TARGET MyApp                 - target 是否存在
# "x" IN_LIST LIST_VAR         - 列表包含

# 逻辑
if(A AND NOT B OR C)
    ...
endif()

# foreach
set(LIST a b c)
foreach(item ${LIST})
    message(STATUS "item=${item}")
endforeach()

foreach(i RANGE 0 5)        # 0..5
    ...
endforeach()

foreach(i RANGE 0 10 2)     # 0,2,4,6,8,10
    ...
endforeach()

# while
set(n 0)
while(n LESS 10)
    math(EXPR n "${n}+1")
endwhile()

5. 函数与宏

# function：有局部 scope，修改形参不影响外面
function(set_xy VAR X Y)
    set(${VAR} "${X};${Y}" PARENT_SCOPE)
    # 不写 PARENT_SCOPE，则 set 仅函数内可见
endfunction()
set_xy(PAIR 1 2)
message(STATUS "${PAIR}")  # 1;2

# macro：没有 scope，相当于文本替换
macro(set_xy_macro VAR X Y)
    set(${VAR} "${X};${Y}")
endmacro()
set_xy_macro(PAIR2 3 4)
message(STATUS "${PAIR2}")  # 3;4

解析参数：

cmake_parse_arguments(
    PREFIX       MY
    OPTIONS      VERBOSE FORCE       # bool 选项
    ONE_VALUE    OUTPUT               # 单值
    MULTI_VALUE  SOURCES              # 多值（列表）
    ${ARGN}                            # 用户的额外参数
)

function(my_add)
    cmake_parse_arguments(MY "${options}" "${oneValueArgs}" "${multiValueArgs}" ${ARGN})

    if(MY_VERBOSE)
        message(STATUS "verbose on")
    endif()
    # MY_OUTPUT, MY_SOURCES 可用了
endfunction()

my_add(VERBOSE OUTPUT out.txt SOURCES a.cpp b.cpp)

6. target_* 命令族（核心）

这是现代 CMake 的"命脉"。每条命令都把属性挂到具体的 target 上，CMake 自己处理传递性，比全局 include_directories() 安全得多。

6.1 PRIVATE / PUBLIC / INTERFACE

关键字含义
PRIVATE仅当前 target 编译时用，链接时不传出
PUBLIC当前 target 用，链接此 target 的人也会继承
INTERFACE自己不用，但链接它的人用
 
# 例：MyLib 有一个私有 include，依赖 fmt 给所有使用者
target_include_directories(MyLib PRIVATE src/)
target_link_libraries(MyLib PUBLIC fmt::fmt)

# App 链接 MyLib 自动获得 fmt 的 include 目录
target_link_libraries(App PRIVATE MyLib)

6.2 INTERFACE 库（header-only 包的推荐写法）

# 给 header-only 库做包裹
add_library(fmt-header INTERFACE)
target_include_directories(fmt-header INTERFACE
    ${fmt_INCLUDE_DIRS}      # 假设 find_package(fmt) 设了这个变量
)
target_compile_definitions(fmt-header INTERFACE FMT_HEADER_ONLY=1)

# 使用时
target_link_libraries(App PRIVATE fmt-header)

6.3 全局命令（不要用）

全局替换为
include_directories(...)target_include_directories(tgt ...)
add_definitions(...)target_compile_definitions(tgt ...)
add_compile_options(...)target_compile_options(tgt ...)
link_libraries(...)target_link_libraries(tgt ...)
全局命令会污染所有 target，包括第三方 find_package 出来的那些，可能导致奇怪的宏定义冲突或 include 路径串台。

7. find_package（核心）

7.1 Module vs CONFIG

模式文件名谁写
ModuleFind<Pkg>.cmakeCMake Modules / vcpkg / 用户自己
CONFIG<Pkg>Config.cmake 或 <pkg>-config.cmake上游项目官方

7.2 怎么确定该用哪种

三种方式：

# 1. 默认：先 Module 再 CONFIG
find_package(Foo REQUIRED)

# 2. 强制 CONFIG（常见于 vcpkg 装的现代库）
find_package(Foo CONFIG REQUIRED)

# 3. 强制 Module（基本不用）
find_package(Foo MODULE REQUIRED)

判断路径：包装了之后，里面有 <Pkg>Config.cmake 才有 CONFIG 模式；有 Find<Pkg>.cmake 才有 Module 模式。

Vulkan SDK 自带 FindVulkan.cmake（Module 模式），所以 find_package(Vulkan) 默认走 Module； vcpkg 装的 glfw3 / glm / tinyobjloader 都用 CONFIG 模式。

7.3 结果变量和 target

find_package 之后会提供两类东西：

# 类型 1：变量
#   Foo_FOUND   - 是否找到
#   Foo_INCLUDE_DIRS / Foo_LIBRARIES / Foo_DEFINITIONS
#   Foo_VERSION / Foo_DIR

# 类型 2：target（最现代）
#   Foo::Foo 或裸 Foo - 包含全部 include + link 信息

优先用 target，干净；如果找不到 target，再退化到变量。

7.4 手写查找（自己 find）

# 找头文件
find_path(MyInc_INCLUDE_DIR myapi.h
    PATHS /opt/myapi/include
    PATH_SUFFIXES myapi
)

# 找库
find_library(MyLib_LIBRARY
    NAMES myapi
    PATHS /opt/myapi/lib
)

find_package_handle_standard_args(MyLib
    REQUIRED_VARS MyInc_INCLUDE_DIR MyLib_LIBRARY
)

if(MyLib_FOUND AND NOT TARGET MyLib::MyLib)
    add_library(MyLib::MyLib UNKNOWN IMPORTED)
    set_target_properties(MyLib::MyLib PROPERTIES
        IMPORTED_LOCATION "${MyLib_LIBRARY}"
        INTERFACE_INCLUDE_DIRECTORIES "${MyInc_INCLUDE_DIR}"
    )
endif()

8. Generator Expressions

8.1 基础

Genex 是写在 $<...> 里的只在生成构建文件时求值的表达式，它能根据 target / 配置 / 平台等动态地改变值。不能在 configure 时当普通变量用。

8.2 常用表达式

表达式含义
$<CONFIG>当前配置（Debug / Release / RelWithDebInfo）
$<CONFIG:Debug>Debug 时为 1，否则为空
$<PLATFORM_ID>Win32 / Linux / Darwin
$<COMPILE_LANGUAGE>CXX / C / CUDA
$<TARGET_FILE:tgt>target 产物路径（含文件名）
$<TARGET_FILE_DIR:tgt>target 产物所在目录
$<TARGET_NAME:tgt>target 名
$<IF:cond,true,false>条件分支（CMake 3.8+）
$<JOIN:list,sep>列表拼接
$<0:...>空（用作占位符）
$<BUILD_INTERFACE:...>仅在 build 树里用
$<INSTALL_INTERFACE:...>仅在 install 树里用
$<SHELL_PATH:...>转换为 shell 路径字符串

8.3 常见坑

# ✅ 用于 target_* 命令的字符串参数里
target_compile_definitions(tgt PRIVATE
    DEBUG_LEVEL=$<IF:$<CONFIG:Debug>,2,0>
)

# ❌ 不用于 set() 里（configure 时求值）
# set(X "$<CONFIG>")          # X 只会得到字面字符串 "$"
# message(STATUS "$<CONFIG>")  # 同样

# ✅ 用于 add_custom_command(COMMAND ...) 里
add_custom_command(TARGET MyApp POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy
        "${VCPKG_INSTALLED_DIR}/${VCPKG_TARGET_TRIPLET}/$<CONFIG>/bin/foo.dll"
        $<TARGET_FILE_DIR:MyApp>
)

# ❌ 嵌套 genex 要 CMake 3.21+ 才稳
# 必要时用中间变量拆开

9. 安装（install）

# 安装可执行文件 / 库 / 头文件
install(TARGETS MyApp MyLib
    RUNTIME DESTINATION bin               # .exe / .dll
    ARCHIVE DESTINATION lib               # .lib (静态)
    LIBRARY DESTINATION lib               # .so / .dylib
    PUBLIC_HEADER DESTINATION include      # header-only 文件
)

install(FILES README.md DESTINATION .)
install(DIRECTORY assets/ DESTINATION share/MyApp)

# install 后用 prefix 改根目录
cmake --install build --prefix C:/myinstall
# 默认 prefix 在 Unix 是 /usr/local，在 Windows 是 C:/Program Files/<project>

10. CMakePresets.json

CMake 3.19+ 引入的配置预设，让跨平台配置可复现。VS 2022 直接读这个文件并列出 preset。

10.1 v6 schema 示例

{
  "version": 6,
  "configurePresets": [
    {
      "name": "vcpkg-debug",
      "displayName": "vcpkg Debug (x64)",
      "generator": "Visual Studio 17 2022",
      "architecture": { "value": "x64", "strategy": "set" },
      "toolset": { "value": "host=x64", "strategy": "set" },
      "toolchainFile": "$env{VCPKG_ROOT}/scripts/buildsystems/vcpkg.cmake",
      "cacheVariables": {
        "CMAKE_BUILD_TYPE": "Debug",
        "MY_OPTION": "ON"
      },
      "environment": {
        "MY_ENV": "value"
      },
      "warnings": {
        "dev": false,
        "deprecated": true
      }
    },
    {
      "name": "vcpkg-release",
      "inherits": "vcpkg-debug",
      "cacheVariables": { "CMAKE_BUILD_TYPE": "Release" }
    }
  ],
  "buildPresets": [
    { "name": "debug",   "configurePreset": "vcpkg-debug" },
    { "name": "release", "configurePreset": "vcpkg-release" }
  ],
  "testPresets": [
    { "name": "default", "configurePreset": "vcpkg-debug" }
  ]
}

常用字段：

字段说明
generator如 Ninja、Visual Studio 17 2022
architecture / toolsetWin32 / x64 / ARM64
toolchainFilevcpkg 一定要写在这里
cacheVariables等价 -DXXX=YYY
environment设置构建时要用的环境变量
inherits从另一个 preset 继承
warnings行为开关（比如关掉 dev warning）

10.2 VS 集成

VS 2022 打开带 CMakePresets.json 的项目，会在解决方案资源管理器上方弹出 "Select a Preset" 下拉。选了之后自动调用 cmake --preset <name>。

VS 默认行为的 Configure 走的是 "default preset"，如果项目里 presets.json 没显式声明，默认行为不会被它接管。你要么选 vs 左上角的 preset，要么自己手动 configure。

11. 平台与编译器差异

# 编译器 ID
if(CMAKE_CXX_COMPILER_ID STREQUAL "GNU")
    # gcc
elseif(CMAKE_CXX_COMPILER_ID STREQUAL "Clang")
    # clang / clang-cl
elseif(CMAKE_CXX_COMPILER_ID STREQUAL "MSVC")
    # MSVC
elseif(CMAKE_CXX_COMPILER_ID STREQUAL "AppleClang")
    ...
endif()

# 平台
if(WIN32)        # 任何 Windows，含 64 位
if(MSVC)         # 编译器是 MSVC
if(CMAKE_SIZEOF_VOID_P EQUAL 8)   # 64 位
if(APPLE) / if(UNIX) / if(LINUX)

# 添加平台专属选项
if(MSVC)
    target_compile_options(MyApp PRIVATE /W4)
endif()

12. add_custom_command 注意点

12.1 configure-time vs build-time

# configure 时执行（生成阶段运行）
add_custom_command(OUTPUT somefile
    COMMAND ${CMAKE_COMMAND} -E echo "configured"
    DEPENDS ...
)
# 注意 OUTPUT 必须存在，否则 CMake 不会触发

# build 时执行（每次 target 构建触发）
add_custom_command(TARGET MyApp POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy x y
)
# 不写 OUTPUT 就是 build 时执行

12.2 多 config 与 $<CONFIG>

VS Generator 是 multi-config：一次生成，Debug / Release 共用同一个项目文件。 Ninja / Make 是 single-config：必须重新 configure 才能换配置。

写 custom command 时：

# ❌ single-config 写法，multi-config 不对
if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    set(BIN_DIR ${VCPKG_INSTALLED_DIR}/${VCPKG_TARGET_TRIPLET}/debug/bin)
else()
    set(BIN_DIR ${VCPKG_INSTALLED_DIR}/${VCPKG_TARGET_TRIPLET}/bin)
endif()

# ✅ multi-config 也能用的写法
add_custom_command(TARGET MyApp POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy
        "${VCPKG_INSTALLED_DIR}/${VCPKG_TARGET_TRIPLET}/$<CONFIG>/bin/foo.dll"
        $<TARGET_FILE_DIR:MyApp>
)

13. 调试技巧

# 输出消息
message(STATUS "普通信息")        # -- 开头
message(WARNING "警告")           # 黄色
message(AUTHOR_WARNING "作者警告") # 黄色，dev 默认显示
message(SEND_ERROR "错误，中止 configure 但不 abort")  # 退出非 0 但不 panic
message(FATAL_ERROR "严重错误")    # 中止一切

# 查看执行的每一行
cmake -S . -B build --trace-expand

# 只看 find_package 的相关调用
cmake -S . -B build --debug-output --trace-expand

# 打印所有 target
cmake --graphviz=foo.dot -S . -B build

# -E 子命令（独立的"小工具"）
cmake -E copy A B
cmake -E copy_if_different A B
cmake -E make_directory DIR
cmake -E echo "hello"
cmake -E remove FILE
cmake -E compare_files A B
cmake -E create_symlink ...

14. 现代 CMake 风格建议

  1. 用 target_* 而不是全局命令
  2. 对每个 target 设置属性，别让所有 target 共享全局设置
  3. PRIVATE / PUBLIC / INTERFACE 区分传播方向
  4. find_package 优先用 target，变量作为 fallback
  5. link 命名空间 target（如 fmt::fmt），没有命名空间就裸用（如 glfw）
  6. PUBLIC_HEADER 配合 install 用，让你的库可以被其他项目复用
  7. CMakeLists.txt 用小写命令（cmake_minimum_required 也是小写加下划线）
  8. 三方库复杂时拆子目录：add_subdirectory(libs/foo) + libs/foo/CMakeLists.txt
  9. 现代 cmake_minimum_required 推荐 3.20+

15. 常见坑

坑原因 & 解法
add_executable 后立刻 link 失败顺序问题：必须先 add target，再 target_link_libraries
find_package(... CONFIG) 找不到包没装 / 装了但没 config 文件。查 installed/<triplet>/share/<pkg>/
改 CMakeLists 后没生效删 build/ 重配：Remove-Item -Recurse build
set(...) 在函数里不生效函数作用域隔离，要么用 PARENT_SCOPE，要么改成 macro
VS 里切换 Debug/Release 配置链接不到对应 dll用了 hard-coded path，没用 $<CONFIG>
find_package(Vulkan) 找不到需要 VULKAN_SDK 环境变量；vcpkg 的 vulkan 包只是 wrapper
target_link_libraries 找不到 IMPORTED targettarget 名字写错了，看 usage 或 grep add_library
option 改了不生效option 把值写到缓存；缓存污染。删 CMakeCache.txt
--trace-expand 输出铺天盖地默认 trace 全部 ; 加 --trace-source=CMakeLists.txt 只看指定文件
